from django.contrib.admin import AdminSite
from django.urls import path
from django.http import JsonResponse
from django.conf import settings
import sys, platform, django


class B2BiAdminSite(AdminSite):
    site_header = "B2Bi Admin Portal"
    site_title  = "B2Bi Staff"
    index_title = "B2Bi Admin Portal"

    def get_urls(self):
        urls = super().get_urls()
        custom = [
            path(
                'system-info/',
                self.admin_view(self.system_info_view),
                name='system_info',
            ),
        ]
        return custom + urls

    def system_info_view(self, request):
        data = {
            "python"         : sys.version,
            "platform"       : platform.platform(),
            "django"         : django.__version__,
            "debug"          : settings.DEBUG,
            "database"       : settings.DATABASES['default']['ENGINE'],
            "database_name"  : str(settings.DATABASES['default']['NAME']),
            "time_zone"      : settings.TIME_ZONE,
            "installed_apps" : list(settings.INSTALLED_APPS),
            "email_host"     : getattr(settings, 'EMAIL_HOST', 'Not set'),
            "email_port"     : getattr(settings, 'EMAIL_PORT', 'Not set'),
            "allowed_hosts"  : settings.ALLOWED_HOSTS,
        }
        return JsonResponse(data, json_dumps_params={'indent': 2})


staff_admin_site = B2BiAdminSite(name="staff_admin")
