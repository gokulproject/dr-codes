# b2bi/models.py

from django.contrib.auth.models import (
    AbstractBaseUser,
    BaseUserManager,
    PermissionsMixin,
)
from django.db import models


# ──────────────────────────────────────────────────────────────────────────────
# B2BI Custom Group
# Table: b2bi_auth_group
# ──────────────────────────────────────────────────────────────────────────────
class B2BIGroup(models.Model):
    name        = models.CharField(max_length=150, unique=True)
    description = models.TextField(blank=True, null=True)
    created_at  = models.DateTimeField(auto_now_add=True)
    updated_at  = models.DateTimeField(auto_now=True)

    def __str__(self):
        return self.name

    class Meta:
        app_label   = "b2bi"
        db_table    = "b2bi_auth_group"          # ← exact table name in DB
        verbose_name        = "B2BI Group"
        verbose_name_plural = "B2BI Groups"
        ordering = ["name"]


# ──────────────────────────────────────────────────────────────────────────────
# B2BI Custom User Manager
# ──────────────────────────────────────────────────────────────────────────────
class B2BIUserManager(BaseUserManager):

    def create_user(self, email, password=None, **extra_fields):
        if not email:
            raise ValueError("Email address is required")
        email = self.normalize_email(email)
        user  = self.model(email=email, **extra_fields)
        user.set_password(password)
        user.save(using=self._db)
        return user

    def create_superuser(self, email, password=None, **extra_fields):
        extra_fields.setdefault("is_staff",      True)
        extra_fields.setdefault("is_superuser",  True)
        extra_fields.setdefault("is_active",     True)

        if not extra_fields.get("is_staff"):
            raise ValueError("Superuser must have is_staff=True")
        if not extra_fields.get("is_superuser"):
            raise ValueError("Superuser must have is_superuser=True")

        return self.create_user(email, password, **extra_fields)


# ──────────────────────────────────────────────────────────────────────────────
# B2BI Custom User
# Table: b2bi_auth_user
# ──────────────────────────────────────────────────────────────────────────────
class B2BIUser(AbstractBaseUser, PermissionsMixin):

    # ── Core Fields ────────────────────────────────────────────────────────────
    email      = models.EmailField(unique=True, verbose_name="Email Address")
    first_name = models.CharField(max_length=100, blank=True)
    last_name  = models.CharField(max_length=100, blank=True)
    phone      = models.CharField(max_length=20,  blank=True, null=True)

    # ── Permission Flags ───────────────────────────────────────────────────────
    is_active    = models.BooleanField(default=True,  verbose_name="Active")
    is_staff     = models.BooleanField(default=False, verbose_name="Staff (b2bi-admin access)")
    is_superuser = models.BooleanField(default=False, verbose_name="Superuser (full access)")

    # ── B2BI Group (custom — not Django's auth group) ──────────────────────────
    b2bi_groups  = models.ManyToManyField(
        B2BIGroup,
        blank=True,
        related_name="users",
        verbose_name="B2BI Groups",
    )

    # ── Timestamps ─────────────────────────────────────────────────────────────
    date_joined = models.DateTimeField(auto_now_add=True)
    updated_at  = models.DateTimeField(auto_now=True)

    # ── Auth Configuration ─────────────────────────────────────────────────────
    USERNAME_FIELD  = "email"
    REQUIRED_FIELDS = []

    objects = B2BIUserManager()

    # ── Helpers ────────────────────────────────────────────────────────────────
    def get_full_name(self):
        return f"{self.first_name} {self.last_name}".strip() or self.email

    def get_short_name(self):
        return self.first_name or self.email

    def __str__(self):
        return self.email

    class Meta:
        app_label           = "b2bi"
        db_table            = "b2bi_auth_user"       # ← exact table name in DB
        verbose_name        = "B2BI User"
        verbose_name_plural = "B2BI Users"
        ordering            = ["-date_joined"]

class B2BIUserAdmin(UserAdmin):
    # ── Forms ──────────────────────────────────────────────────────────────────
    add_form = B2BIUserCreationForm
    form     = B2BIUserChangeForm
    model    = B2BIUser

    # ── List View ──────────────────────────────────────────────────────────────
    list_display = [
        "email",
        "get_full_name",
        "phone",
        "is_active_badge",
        "is_staff_badge",
        "is_superuser_badge",
        "group_list",
        "date_joined",
    ]
    list_display_links = ["email", "get_full_name"]
    list_filter        = ["is_active", "is_staff", "is_superuser", "b2bi_groups", "date_joined"]
    search_fields      = ["email", "first_name", "last_name", "phone"]
    ordering           = ["-date_joined"]
    list_per_page      = 25
    date_hierarchy     = "date_joined"
    filter_horizontal  = ("b2bi_groups",)

    # ── Coloured badges for boolean fields ────────────────────────────────────
    @admin.display(description="Active", ordering="is_active")
    def is_active_badge(self, obj):
        color = "#28a745" if obj.is_active else "#dc3545"
        label = "Yes" if obj.is_active else "No"
        return format_html(
            '<span style="color:{}; font-weight:bold;">{}</span>', color, label
        )

    @admin.display(description="Staff", ordering="is_staff")
    def is_staff_badge(self, obj):
        color = "#007bff" if obj.is_staff else "#6c757d"
        label = "Yes" if obj.is_staff else "No"
        return format_html(
            '<span style="color:{}; font-weight:bold;">{}</span>', color, label
        )

    @admin.display(description="Superuser", ordering="is_superuser")
    def is_superuser_badge(self, obj):
        color = "#fd7e14" if obj.is_superuser else "#6c757d"
        label = "Yes" if obj.is_superuser else "No"
        return format_html(
            '<span style="color:{}; font-weight:bold;">{}</span>', color, label
        )

    @admin.display(description="Groups")
    def group_list(self, obj):
        groups = obj.b2bi_groups.all()
        if not groups:
            return format_html('<span style="color:#aaa;">—</span>')
        badges = " ".join(
            f'<span style="background:#e9ecef; padding:2px 8px; border-radius:10px; '
            f'font-size:11px;">{g.name}</span>'
            for g in groups
        )
        return format_html(badges)

    # ── CHANGE User form layout ────────────────────────────────────────────────
    fieldsets = (
        (_("Login Credentials"), {
            "fields": ("email", "password"),
        }),
        (_("Personal Information"), {
            "fields": ("first_name", "last_name", "phone"),
        }),
        (_("Permissions & Access"), {
            "fields": ("is_active", "is_staff", "is_superuser"),
            "description": (
                "<b>is_staff</b> → Can login to /b2bi-admin/ &nbsp;|&nbsp; "
                "<b>is_superuser</b> → Full access with no restrictions"
            ),
        }),
        (_("B2BI Groups"), {
            "fields": ("b2bi_groups",),
        }),
        (_("Important Dates"), {
            "fields": ("last_login", "date_joined"),
            "classes": ("collapse",),
        }),
    )
    readonly_fields = ("last_login", "date_joined")

    # ── ADD User form layout ───────────────────────────────────────────────────
    add_fieldsets = (
        (_("Account Credentials"), {
            "classes": ("wide",),
            "fields": ("email", "password1", "password2"),
        }),
        (_("Personal Information"), {
            "classes": ("wide",),
            "fields": ("first_name", "last_name", "phone"),
        }),
        (_("Permissions & Access"), {
            "classes": ("wide",),
            "fields": ("is_active", "is_staff", "is_superuser"),
        }),
        (_("B2BI Groups"), {
            "classes": ("wide",),
            "fields": ("b2bi_groups",),
        }),
    )

    # ── Bulk Actions ───────────────────────────────────────────────────────────
    actions = ["activate_users", "deactivate_users", "make_staff", "remove_staff"]

    @admin.action(description="✅ Activate selected users")
    def activate_users(self, request, queryset):
        updated = queryset.update(is_active=True)
        self.message_user(request, f"{updated} user(s) activated successfully.")

    @admin.action(description="🚫 Deactivate selected users")
    def deactivate_users(self, request, queryset):
        updated = queryset.update(is_active=False)
        self.message_user(request, f"{updated} user(s) deactivated.")

    @admin.action(description="🔑 Grant staff access")
    def make_staff(self, request, queryset):
        updated = queryset.update(is_staff=True)
        self.message_user(request, f"{updated} user(s) granted staff access.")

    @admin.action(description="🔒 Remove staff access")
    def remove_staff(self, request, queryset):
        updated = queryset.update(is_staff=False)
        self.message_user(request, f"{updated} user(s) staff access removed.")

# User Forms
# ──────────────────────────────────────────────────────────────────────────────

class B2BIUserCreationForm(forms.ModelForm):
    """Form used when ADDING a new B2BIUser in /b2bi-admin/"""

    password1 = forms.CharField(
        label="Password",
        widget=forms.PasswordInput(attrs={"autocomplete": "new-password"}),
        strip=False,
    )
    password2 = forms.CharField(
        label="Confirm Password",
        widget=forms.PasswordInput(attrs={"autocomplete": "new-password"}),
        strip=False,
        help_text="Enter the same password as above for verification.",
    )

    class Meta:
        model  = B2BIUser
        fields = (
            "email",
            "first_name",
            "last_name",
            "phone",
            "is_active",
            "is_staff",
            "is_superuser",
            "b2bi_groups",
        )

    def clean_password2(self):
        p1 = self.cleaned_data.get("password1")
        p2 = self.cleaned_data.get("password2")
        if p1 and p2 and p1 != p2:
            raise ValidationError("The two passwords do not match.")
        return p2

    def save(self, commit=True):
        user = super().save(commit=False)
        user.set_password(self.cleaned_data["password1"])   # hashes correctly
        if commit:
            user.save()
            self.save_m2m()
        return user


class B2BIUserChangeForm(forms.ModelForm):
    """Form used when CHANGING/UPDATING an existing B2BIUser in /b2bi-admin/"""

    password = ReadOnlyPasswordHashField(
        label="Password",
        help_text=(
            "Raw passwords are not stored. Use "
            '<a href="../password/">this form</a> to change it.'
        ),
    )

    class Meta:
        model  = B2BIUser
        fields = (
            "email",
            "first_name",
            "last_name",
            "phone",
            "password",
            "is_active",
            "is_staff",
            "is_superuser",

# b2bi/admin_site.py

from django.contrib.admin import AdminSite


class B2BIAdminSite(AdminSite):
    # ── Branding ───────────────────────────────────────────────────────────────
    site_header       = "B2BI Administration"
    site_title        = "B2BI Admin Portal"
    index_title       = "B2BI Admin Dashboard"
    site_url          = None   # removes "View Site" link from top bar

    # ── Access Control ─────────────────────────────────────────────────────────
    def has_permission(self, request):
        """
        Only B2BIUser with is_staff=True can enter /b2bi-admin/.
        Django's default auth.User is completely BLOCKED from this panel.
        """
        from b2bi.models import B2BIUser

        return (
            request.user.is_authenticated
            and isinstance(request.user, B2BIUser)   # must be B2BIUser (not auth.User)
            and request.user.is_active
            and request.user.is_staff
        )


# ── Singleton — import this everywhere ────────────────────────────────────────
b2bi_admin_site = B2BIAdminSite(name="b2bi_admin")


# b2bi/backends.py

from b2bi.models import B2BIUser


class B2BIAuthBackend:
    """
    Authenticates only B2BIUser instances (email + password).
    Used exclusively for /b2bi-admin/ login.
    Django's default auth.User is NOT authenticated through this backend.
    """

    def authenticate(self, request, username=None, password=None, **kwargs):
        # username field on the login form maps to email (USERNAME_FIELD = 'email')
        email = username or kwargs.get("email")
        if not email:
            return None
        try:
            user = B2BIUser.objects.get(email=email)
            if user.check_password(password) and user.is_active:
                return user
        except B2BIUser.DoesNotExist:
            return None

    def get_user(self, user_id):
        try:
            return B2BIUser.objects.get(pk=user_id)
        except B2BIUser.DoesNotExist:
            return None

## Setup Steps

### 1. Add app to settings
```python
INSTALLED_APPS = [
    ...
    "b2bi",
]

AUTHENTICATION_BACKENDS = [
    "django.contrib.auth.backends.ModelBackend",
    "b2bi.backends.B2BIAuthBackend",
]
```

### 2. Add URL
```python
# myproject/urls.py
from b2bi.admin_site import b2bi_admin_site

urlpatterns = [
    path("admin/",      admin.site.urls),
    path("b2bi-admin/", b2bi_admin_site.urls),
]
```

### 3. Run migrations
```bash
python manage.py makemigrations b2bi
python manage.py migrate
```

### 4. Create a B2BI superuser (for /b2bi-admin/)
```bash
python manage.py shell
```
```python
from b2bi.models import B2BIUser

B2BIUser.objects.create_superuser(
    email    = "admin@b2bi.com",
    password = "securepassword123"
)
```

### 5. Run server
```bash
python manage.py runserver
```

## Features in /b2bi-admin/

### B2BI Users
- List with: email, full name, phone, active/staff/superuser badges, groups, joined date
- Search by: email, first name, last name, phone
- Filter by: active, staff, superuser, groups, date joined
- Bulk actions: Activate, Deactivate, Grant Staff, Remove Staff
- Add: email + password + personal info + permissions + groups
- Change: all fields + hashed password link

### B2BI Groups
- List with: name, description, user count, created/updated dates
- Search by: name, description
- Filter by: created date
- Add/Change: name + description

## Access Rules

```
B2BIUser (is_staff=True)   → /b2bi-admin/ ✅   /admin/ ❌
B2BIUser (is_staff=False)  → /b2bi-admin/ ❌   /admin/ ❌
auth.User (superuser)      → /admin/      ✅   /b2bi-admin/ ❌
```
